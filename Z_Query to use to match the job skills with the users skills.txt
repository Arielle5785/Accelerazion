SELECT 
    u.id AS user_id, 
    u.name AS user_name, 
    COUNT(js.skill_id) AS matching_skills
FROM 
    job_skills js
JOIN 
    users_skills us ON js.skill_id = us.skill_id -- Match skills between jobs and users
JOIN 
    users u ON us.user_id = u.id -- Get user details
JOIN 
    users_class uc ON u.id = uc.user_id -- Ensure the user is a mentee
WHERE 
    js.job_id = <job_id> -- Replace <job_id> with the specific job ID
    AND uc.type_id = 1 -- Ensure the user is a mentee
GROUP BY 
    u.id, u.name
ORDER BY 
    matching_skills DESC -- Sort by the number of matching skills
LIMIT 10; -- Get the top 10 users
